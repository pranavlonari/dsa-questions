Medium DFSGraphArrayStringStack

  1. Handle zero: If num == 0, return "Zero".
  2. Define arrays for constant mappings:

  1. Process in groups of 3 digits (from least significant):

‚úÖ Java implementation (clean & commented)

public class NumberToWords {
    // Constants for mapping
    private static final String[] below_20 = {
        "", "One", "Two", "Three", "Four", "Five", "Six", "Seven", "Eight", "Nine", "Ten",
        "Eleven", "Twelve", "Thirteen", "Fourteen", "Fifteen", "Sixteen", "Seventeen", "Eighteen", "Nineteen"
    };

    private static final String[] tens = {
        "", "", "Twenty", "Thirty", "Forty", "Fifty", "Sixty", "Seventy", "Eighty", "Ninety"
    };

    private static final String[] thousands = {"", "Thousand", "Million", "Billion"};

    public String numberToWords(int num) {
        if (num == 0) return "Zero";

        int i = 0;
        String words = "";

        // Process in groups of 3 digits
        while (num > 0) {
            int chunk = num % 1000;
            if (chunk != 0) {
                words = helper(chunk) + thousands[i] + (words.isEmpty() ? "" : " " + words);
            }
            num /= 1000;
            i++;
        }

        return words.trim();
    }

    // Convert number < 1000 to English words
    private String helper(int num) {
        if (num == 0) return "";
        else if (num < 20) return below_20[num] + " ";
        else if (num < 100) {
            return tens[num / 10] + " " + helper(num % 10);
        } else {
            return below_20[num / 100] + " Hundred " + helper(num % 100);
        }
    }

    // Quick tests
    public static void main(String[] args) {
        NumberToWords converter = new NumberToWords();

        System.out.println(converter.numberToWords(123));        // One Hundred Twenty Three
        System.out.println(converter.numberToWords(12345));      // Twelve Thousand Three Hundred Forty Five
        System.out.println(converter.numberToWords(1234567));    // One Million Two Hundred Thirty Four Thousand Five Hundred Sixty Seven
        System.out.println(converter.numberToWords(1000000));    // One Million
        System.out.println(converter.numberToWords(0));          // Zero
    }
}
